Consider the following algorithm:

schedule([x1, x2, ..., xn]):
  if [x1, ..., xn] empty:
    return []
  else:
    select x1
    // make a list that has the elements of [x2, x3, ..., xn] that
    //   don't interfere with x1
    remaining = [xi, xj, ..., xp]
    // find the maximal schedule on noninterfering
    recur = schedule(noninterfering)
    // prepend (:) x1 to the recursive result
    return x1 : recur
